int n;
scanf("%d", &n);
int arr[n];
for(int i=0;i<n;++i){
	scanf("%d", &arr[i]);
}
for(int i=1;i<n;++i){
	arr[i]+=arr[i-1];
}
int m;
scanf("%d", &m);
for (int i=0;i<m;++i){
	int ini,ult;
	scanf("%d %d", &ini, &ult);
	int result=0;
	for(int i = ini;i <= ult;i++){
		int atras = arr[ini - 1];
		if(ini == 0){
			atras = 0;
		}
		result = arr[ult] - atras;
	} 
    printf("%d\n", result);
}
